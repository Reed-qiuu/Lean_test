\section{Introduction}

The \emph{Chinese Remainder Theorem} (CRT) is one of the foundational
results of elementary number theory. In its classical form, it asserts
that a system of simultaneous congruences with pairwise coprime moduli
admits a unique solution modulo the product of those moduli. In modern
ring-theoretic language, CRT identifies $\mathbb{Z}/N\mathbb{Z}$ with
the product $\prod_i \mathbb{Z}/n_i\mathbb{Z}$ whenever the $n_i$ are
pairwise coprime and $N = \prod_i n_i$.

This blueprint formalizes the two-modulus version of CRT (the heart of
the result, from which the general case follows by induction) together
with its key supporting lemma about congruences modulo a product of
coprime integers, and then states the general $n$-modulus form as a
ring isomorphism.

\section{Supporting Lemma}

\begin{lemma}[Congruence modulo a product of coprime integers]
    \label{lem:modEq-mul-coprime}
    Let $m, n$ be coprime natural numbers and let $a, b$ be natural
    numbers. Then
    \[
        a \equiv b \pmod{m} \;\text{ and }\; a \equiv b \pmod{n}
        \quad\Longleftrightarrow\quad
        a \equiv b \pmod{m\,n}.
    \]
\end{lemma}
\begin{proof}
    The forward direction follows from the fact that if $m \mid (a-b)$
    and $n \mid (a-b)$, then their least common multiple divides
    $a-b$; since $\gcd(m,n) = 1$, we have $\operatorname{lcm}(m,n) =
    mn$, hence $mn \mid (a-b)$. The reverse direction is immediate:
    if $mn \mid (a-b)$ then both $m \mid (a-b)$ and $n \mid (a-b)$.
\end{proof}

\section{Two-Modulus Chinese Remainder Theorem}

\begin{lemma}[CRT for two coprime moduli]
    \label{lem:crt-two}
    \uses{lem:modEq-mul-coprime}
    Let $m, n$ be natural numbers with $\gcd(m, n) = 1$. Then the
    natural ring homomorphism
    \[
        \mathbb{Z}/(mn)\mathbb{Z} \longrightarrow
        \mathbb{Z}/m\mathbb{Z} \times \mathbb{Z}/n\mathbb{Z}
    \]
    induced by the quotient maps is a ring isomorphism.
\end{lemma}
\begin{proof}
    \uses{lem:modEq-mul-coprime}
    Define the forward map as the canonical ring homomorphism sending
    $x \bmod mn$ to $(x \bmod m,\, x \bmod n)$; this is well-defined
    because $m \mid mn$ and $n \mid mn$.

    \emph{Injectivity.} If $x \equiv y \pmod{m}$ and $x \equiv y
    \pmod{n}$, then by Lemma~\ref{lem:modEq-mul-coprime} we have $x
    \equiv y \pmod{mn}$.

    \emph{Surjectivity.} Given a pair $(a \bmod m,\, b \bmod n)$, we
    must produce $x$ with $x \equiv a \pmod{m}$ and $x \equiv b
    \pmod{n}$. Since $\gcd(m,n) = 1$, B\'ezout's identity provides
    integers $u, v$ with $u m + v n = 1$. Set $x = a v n + b u m$.
    Then $x \equiv a v n \equiv a \cdot 1 \equiv a \pmod{m}$ and
    similarly $x \equiv b \pmod{n}$.

    Both sides are finite of size $mn$ (when $mn \neq 0$), so
    injectivity together with equal cardinalities gives a bijection.
    The map is a ring homomorphism by construction, so it is a ring
    isomorphism.
\end{proof}

\section{General Chinese Remainder Theorem}

\begin{theorem}[Chinese Remainder Theorem]
    \label{thm:crt}
    \uses{lem:crt-two}
    Let $\iota$ be a finite index type and let $(n_i)_{i \in \iota}$
    be a family of natural numbers that are pairwise coprime, i.e.\
    $\gcd(n_i, n_j) = 1$ for all $i \neq j$. Set $N = \prod_{i \in
    \iota} n_i$. Then the natural ring homomorphism
    \[
        \mathbb{Z}/N\mathbb{Z} \;\longrightarrow\;
        \prod_{i \in \iota} \mathbb{Z}/n_i\mathbb{Z}
    \]
    is a ring isomorphism. In particular, for every tuple $(a_i)_{i
    \in \iota}$ of integers, the system of congruences
    \[
        x \equiv a_i \pmod{n_i}, \qquad i \in \iota,
    \]
    has a solution $x \in \mathbb{Z}$, and any two solutions differ
    by a multiple of $N$.
\end{theorem}
\begin{proof}
    \uses{lem:crt-two, lem:modEq-mul-coprime}
    The two-modulus case is Lemma~\ref{lem:crt-two}. The general case
    follows by induction on the cardinality of $\iota$.

    The base cases $|\iota| = 0$ and $|\iota| = 1$ are immediate: when
    $\iota$ is empty the empty product is $1$ and both sides are
    trivial; when $\iota$ is a singleton both sides agree on the
    nose.

    For the inductive step, suppose $|\iota| = k+1$. Pick an element
    $i_0 \in \iota$ and set $\iota' = \iota \setminus \{i_0\}$, $m =
    n_{i_0}$, $n = \prod_{i \in \iota'} n_i$. Pairwise coprimality
    implies that $m$ is coprime to $n$ (since $m$ is coprime to each
    factor of $n$, hence to their product). By
    Lemma~\ref{lem:crt-two},
    \[
        \mathbb{Z}/(mn)\mathbb{Z} \;\cong\;
        \mathbb{Z}/m\mathbb{Z} \times \mathbb{Z}/n\mathbb{Z}.
    \]
    By the induction hypothesis applied to $(n_i)_{i \in \iota'}$,
    \[
        \mathbb{Z}/n\mathbb{Z} \;\cong\; \prod_{i \in \iota'}
        \mathbb{Z}/n_i\mathbb{Z}.
    \]
    Composing these isomorphisms yields the desired isomorphism for
    $\iota$, since $N = m n$.

    The statement about systems of congruences is exactly the
    surjectivity (existence) and injectivity (uniqueness modulo $N$)
    of the displayed ring homomorphism.
\end{proof}
